<?php
$con = mysqli_connect('localhost', 'root', 'convox32', 'convoxccs32');

if (!$con) {
    die("Connection failed: " . mysqli_connect_error());
}

$where_conditon = "list_id='ManualList' AND lead_status='NEW' AND process_name='Convox_Progressive_Zoho' AND tobe_dialed='N' AND mobile_number!=''";
$query = "SELECT lead_id FROM convoxccs_list WHERE $where_conditon";
//echo $query. "<br>";
//exit;
$placed_calls_rslt = mysqli_query($con, $query);
if ($placed_calls_rslt) {
    $dialing_lead_ids = [];
    while ($placed_row = mysqli_fetch_assoc($placed_calls_rslt)) {
        $dialing_lead_ids[] = $placed_row["lead_id"];
    }
    $lead_ids_str = implode(',', $dialing_lead_ids);
    $update_query = "UPDATE convoxccs_list SET tobe_dialed='Y' WHERE $where_conditon  AND lead_id IN ($lead_ids_str)";
//echo $update_query;
//exit;
    if (mysqli_query($con, $update_query)) {
        echo "Records updated successfully.";
    } else {
        echo "Error updating records: " . mysqli_error($con);
    }
} else {
    echo "Error in query: " . mysqli_error($con);
}
mysqli_close($con);
?>
