		function unCompressFile($recording_file_path)
		{
			$result = array();
			if (!empty($recording_file_path)) {
				$unixtime = time();
				$recording_file_array = explode(".", $recording_file_path);
				$recording_file = $recording_file_array[0];
				$recording_format = $recording_file_array[1];
				$unixtime = time();
				$src_recording_file = $recording_file . "." . $recording_format;
				$dst_recording_file = $recording_file . "_" . $unixtime . "." . $recording_format;
				$sox_command = "sox $src_recording_file -c 1 -r 3k -b 16 $dst_recording_file";
				system($sox_command);
				if (file_exists($dst_recording_file)) {
					system("mv -f $dst_recording_file $src_recording_file");
				}
			}
			return $result;
		}
