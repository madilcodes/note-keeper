1 if callrecording sample size is 8000 , need to reduce the sample rate 3000 (manually )then 
2 add this file in root audio.sh
3 give the path 2024-01-13  in audio.sh file   , it will convert gsm and 8000 to wav,3000 .
4 then run the file sh audio.sh

#!/bin/bash
path="/var/www/html/calls/2024-01-18/out/"
cd "$path" || exit

for src_recording_file in *; do
    if [ -f "$src_recording_file" ]; then
        dst_recording_file="temp_$src_recording_file"


filename="$src_recording_file"

if [ ! -f "$filename" ]; then
    echo "File '$filename' not found."
    exit 1
fi

soxi_output=$(soxi "$filename")
sample_encoding=$(echo "$soxi_output" | grep "Sample Encoding" | awk -F ': ' '{print $2}')
sample_rate=$(echo "$soxi_output" | grep "Sample Rate" | awk -F ': ' '{print $2}')

# Storing Sample Encoding and Sample Rate in variables
sample_encoding_variable="$sample_encoding"
sample_rate_variable="$sample_rate"

if [ "$sample_encoding_variable" == "GSM" ] && [ "$sample_rate_variable" == "8000" ]; then
 sox_command="sox $src_recording_file -c 1 -r 3k -b 16 $dst_recording_file"
        $sox_command


        if [ -e "$dst_recording_file" ]; then
            mv -f "$dst_recording_file" "$src_recording_file"
            echo "Processed: $src_recording_file"
        else
            echo "Error processing: $src_recording_file"
        fi

    echo "hello working  $src_recording_file"




fi
    fi
        done
